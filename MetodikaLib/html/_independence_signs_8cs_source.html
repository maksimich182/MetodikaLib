151 private double _getPk(MarkTable markTable)
152 {
153 ProcessChanged?.Invoke(this, $"Рассчет массивов u и v для k = {markTable.Dimension}");
154 int dimensionV = markTable.Dimension - 1;
155 double Pk;
156 long[] Vs = new long[1 << dimensionV];
157 long[] Us = new long[2];
158
159 for (int i = 0; i < Vs.Length; i++)
160 {
161 Vs[i] = markTable.Table[i * 2] + markTable.Table[i * 2 + 1];
162 Us[0] += markTable.Table[i * 2];
163 Us[1] += markTable.Table[i * 2 + 1];
164 ProgressChanged?.Invoke(this, Tools.GetPercent(i + 1, Vs.Length));
165 }
166
167
168 double dStatistic = _getStatistic(markTable, Vs, Us);
169 _statistics.Add(dStatistic);
170 Pk = 1 - ChiSquared.CDF((1 << dimensionV) - 1, dStatistic);
171 _pValues.Add(Pk);
172
173 return Pk;
174 }

183 private double _getStatistic(MarkTable markTable, long[] Vs, long[] Us)
184 {
185 ProcessChanged?.Invoke(this, $"Рассчет статистики для k = {markTable.Dimension}");
186 double statistic = 0;
187 long bufVector = 0;
188
189 BigInteger firstProduct;
190 BigInteger secondProduct;
191 BigInteger elementSquare;
192
193 for (int i = 0; i < Vs.Length; i++)
194 {
195 bufVector = i << 1;
196 for (int j = 0; j < 2; j++)
197 {
198 firstProduct = (BigInteger)markTable.NmVectors * markTable.Table[bufVector];
199 secondProduct = (BigInteger)Vs[i] * Us[j];
200 elementSquare = firstProduct - secondProduct;
201 statistic += Tools.BigIntDevider((BigInteger.Pow(firstProduct - secondProduct, 2)), secondProduct);
202 bufVector |= 1;
203 }
204 ProgressChanged?.Invoke(this, Tools.GetPercent(i + 1, Vs.Length));
205 }
206 statistic /= markTable.NmVectors;
207
208 return statistic;
209 }
